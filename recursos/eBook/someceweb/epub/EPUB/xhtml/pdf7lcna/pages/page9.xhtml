Código para la gráfica de compración de la distribución binomial con diferentes parámetros
rm(list=ls())
par(mfrow=c(2,2))
plot(0:20, dbinom(0:20, 20, 0.5), type="h", col=4, lwd=3,
main="Binomial (n = 20, p = 0.5)", ylab="probabilidad",
xlab="figura 1")
plot(0:20, dbinom(0:20, 20, 0.7), type="h", col=4, lwd=3,
main="Binomial (n = 20, p = 0.7)", ylab="probabilidad",
xlab="figura 2")
plot(0:20, dbinom(0:20, 20, 0.3), type="h", col=4, lwd=3,
main="Binomial (n = 20, p = 0.3)", ylab="probabilidad",
xlab="figura 3")
plot(0:10, dbinom(0:10, 10, 0.5), type="h", col=4, lwd=3,
main="Binomial (n = 10, p = 0.5)", ylab="probabilidad",
xlab="figura 4")
