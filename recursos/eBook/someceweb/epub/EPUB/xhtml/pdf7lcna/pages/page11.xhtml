Código para las gráficas de comparación de las distribuciones normal y t de distribución
rm(list=ls())
x <- seq(-4, 4, length=100)
hx <- dnorm(x)
par(mfrow=c(2,2))
plot(x, hx, type="l", lty=2, col=4, lwd=3, xlab="1 grado de libertad", ylab="densidad",
main="Normal vs distibución t")
lines(x, dt(x, 1), lwd=2, col=2)
plot(x, hx, type="l", lty=2, col=4, lwd=3, xlab="3 grados de libertad", ylab="densidad",
main="Normal vs distibución t")
lines(x, dt(x, 3), lwd=2, col=2)
plot(x, hx, type="l", lty=2, col=4, lwd=3, xlab="8 grados de libertad", ylab="densidad",
main="Normal vs distibución t")
lines(x, dt(x, 8), lwd=2, col=2)
plot(x, hx, type="l", lty=2, col=4, lwd=3, xlab="24 grados de libertad", ylab="densidad",
main="Normal vs distibución t")
lines(x, dt(x, 24), lwd=2, col=2)
par(mfrow=c(1,1))
